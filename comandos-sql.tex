create table requester(
	id uuid not null primary key,
	name varchar(40) not null,
	email varchar(40) not null,
	department varchar(40) not null,
	build numeric(3) not null,
	constraint chk_department check (department in('QUALITY_CENTER', 'AUTOMOTIVE', 'MARKETING', 'HEALTH', 'ENGINEERING', 'ABRASIVES'))
);

create table sample(
	id uuid not null primary key,
	commodity numeric(15) not null,
	description varchar(30) not null,
	price numeric(14,2) not null,
	status varchar(15) not null,
	genero varchar(15) not null,
	id_requester uuid not null references requester(id),
	constraint chk_status check(status in ('PENDING', 'IN_TRANSIT', 'RECEIVED_BY_PLANT_WAREHOUSE', 'DELIVERED', 'RECEIVED_BY_REQUESTER'))
);
