create table requester(
	id uuid not null primary key,
	name varchar(40) not null,
	email varchar(40) not null,
	department varchar(40) not null,
	build numeric(3) not null
);

create table sample(
	id uuid not null primary key,
	commodity numeric(15) not null,
	description varchar(30) not null,
	price numeric(14,2) not null,
	status varchar(15) not null,
	genero varchar(15) not null,
	id_autor uuid not null references requester(id),
	constraint chk_genero check(genero in ('PENDING', 'IN_TRANSIT', 'RECEIVED_BY_PLANT_WAREHOUSE', 'DELIVERED', 'RECEIVED_BY_REQUESTER'))
);
